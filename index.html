sticky on table

안내
  • css postion sticky 만 사용해서 제어
  • 큰 테이블 테스트용
설명
  • attribute
  • class="sot" //기본 style 클래스명
  • data-sot-top="2" //상단 고정 row 수
  • data-sot-left="2" //좌측 고정 col 수
  • <div class="sot" data-sot-top="2" data-sot-left="2" style="width:300px; height:300px; margin:0 auto;">
        <table> .... </table>
    </div>
  • script
  • StickyOnTable.apply(대상); //sticky on table 를 적용한다.
    StickyOnTable.apply(document.querySelector("#sot6"));
  • StickyOnTable.reset(대상); //sticky on table 를 해제한다.
    StickyOnTable.reset(document.querySelector("#sot6"));
